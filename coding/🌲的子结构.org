class Solution:
    def isSubStructure(self, A: TreeNode, B: TreeNode) -> bool:
        if not B or not A:
            return False
        def dfs(A, B):
            if not B:
                return True
            if not A:
                return False
            if A.val == B.val:
                return dfs(A.left, B.left) and dfs(A.right, B.right)
            else:
                return False
        queue = [A]
        while queue:
            queue_len = len(queue)
            for i in range(queue_len):
                node = queue.pop(0)
                if dfs(node, B):
                    return True
                if node.left:
                    queue.append(node.left)
                if node.right:
                    queue.append(node.right)
        return False
