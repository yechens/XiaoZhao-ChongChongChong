class Solution:
    def pathSum(self, root: TreeNode, sum: int) -> int:
        ans = 0
        if not root:
            return ans
        def dfs(root, s, path):
            nonlocal ans
            s_ = s + root.val
            path_ = path + [root]
            if s_ == sum:
                ans += 1
            if root.left:
                dfs(root.left, s_, path_)
            if root.right:
                dfs(root.right, s_, path_)
        queue = [root]
        while queue:
            queue_len = len(queue)
            for i in range(queue_len):
                node = queue.pop(0)
                dfs(node, 0, [])
                if node.left:
                    queue.append(node.left)
                if node.right:
                    queue.append(node.right)
        return ans
